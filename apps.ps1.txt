# -------- CENTRALIZAR TEXTO --------

function Write-Centered {

param($text,$color="White")

$consoleWidth = 90
$windowWidth = $Host.UI.RawUI.WindowSize.Width

$padding = [math]::Max(0,($consoleWidth-$text.Length)/2)
$sideSpace = [math]::Max(0,($windowWidth-$consoleWidth)/2)

Write-Host (" " * [int]($sideSpace + $padding) + $text) -ForegroundColor $color

}

# -------- LOGIN --------

function Sistema-Login {

$usuarioCorreto = "admin"
$senhaCorreta = "1234"

$tentativas = 0
$maxTentativas = 3

while ($tentativas -lt $maxTentativas) {

Clear-Host
Linha
Write-Centered "ACESSO RESTRITO - COLEGIO CRUZEIRO TI" Cyan
Linha

$usuario = Read-Host "sbht-ti"
$senha = Read-Host "infra@2026"

if ($usuario -eq $usuarioCorreto -and $senha -eq $senhaCorreta) {

Write-Centered "Login realizado com sucesso!" Green
Start-Sleep 1
Menu-Principal
return

}

else {

$tentativas++

Write-Centered "Usuario ou senha incorretos!" Red
Write-Centered "Tentativa $tentativas de $maxTentativas" Yellow
Start-Sleep 2

}

}

Write-Centered "ACESSO BLOQUEADO!" Red
Start-Sleep 2
Exit

}

# -------- INSTALAR APPS --------

function Instalar-Apps {

param($apps)

foreach ($app in $apps) {

Write-Centered "Instalando $($app.nome)..." Cyan

$result = winget list --id $($app.id) -e 2>$null | Out-String

if ($result -notmatch $app.id) {

winget install -e --id $($app.id) `
--accept-package-agreements `
--accept-source-agreements `
--silent

Write-Centered "$($app.nome) instalado" Green

}
else {

Write-Centered "$($app.nome) ja instalado" Yellow

}

}

Pause

}

# -------- BITDEFENDER --------

function Instalar-Bitdefender {

$url = "COLOQUE_AQUI_LINK_BITDEFENDER"
$dest = "$env:TEMP\bitdefender.exe"

Write-Centered "Baixando Bitdefender..." Yellow

Invoke-WebRequest $url -OutFile $dest

Write-Centered "Instalando Bitdefender..." Yellow

Start-Process $dest -ArgumentList "/silent" -Wait

Pause

}

# -------- OFFICE 2016 --------

function Instalar-Office2016 {

$officePath = "$env:TEMP\Office2016"
New-Item -ItemType Directory -Force -Path $officePath | Out-Null

Write-Centered "Baixando Office Deployment Tool..." Yellow

$url = "https://download.microsoft.com/download/0/2/A/02A7D4D6-7D2C-4E36-9A8B-2E8D9D4F9F9F/officedeploymenttool.exe"
$odt = "$officePath\odt.exe"

Invoke-WebRequest $url -OutFile $odt

Start-Process $odt -ArgumentList "/quiet /extract:$officePath" -Wait

Write-Centered "Criando configuracao Office 2016..." Yellow

$xml = @"
<Configuration>
  <Add OfficeClientEdition="64" Channel="PerpetualVL2016">
    <Product ID="ProPlusVolume">
      <Language ID="pt-br"/>
    </Product>
  </Add>
  <Display Level="None" AcceptEULA="TRUE"/>
</Configuration>
"@

$xmlPath = "$officePath\config.xml"
$xml | Out-File -Encoding UTF8 $xmlPath

Write-Centered "Baixando Office 2016..." Cyan
Start-Process "$officePath\setup.exe" -ArgumentList "/download $xmlPath" -Wait

Write-Centered "Instalando Office 2016..." Green
Start-Process "$officePath\setup.exe" -ArgumentList "/configure $xmlPath" -Wait

Write-Centered "Office 2016 instalado com sucesso!" Green

Pause

}
